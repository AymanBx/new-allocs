27//----------------------------------------------------------------------
28// Utility function for memory swapping
29// Used because TGA stores the RGB data in reverse order (BGR)
30//----------------------------------------------------------------------
31void swapRGB(unsigned char* theData, int height, int width)
32{
33 int imgSize;
34 unsigned char tmp;
35
36 imgSize = height*width;
37
38 for(int k = 0; k < imgSize; k++)
39 {
40 tmp = theData[k*3+2];
41 theData[k*3+2] = theData[k*3];
42 theData[k*3] = tmp;
43 }
44}

60// ---------------------------------------------------------------------
61// Function : readTGA
62// Description :
63//
64// This function reads an image of type TGA (8 or 24 bits, uncompressed
65//
66//----------------------------------------------------------------------
67
68RasterImage readTGA(const char* filePath)
69{
70 RasterImage info;
71
72 //--------------------------------
73 // open TARGA input file
74 //--------------------------------
75 FILE* tga_in = fopen(filePath, "rb" );
76 if (tga_in == NULL)
77 {
78 printf("Cannot open image file %s\n", filePath);
79 exit(11);
80 }
81
82 //--------------------------------
83 // Read the header (TARGA file)
84 //--------------------------------
85 char head[18] ;
86 fread( head, sizeof(char), 18, tga_in ) ;
87 /* Get the size of the image */
88 info.width = (int)(((unsigned int)head[12]&0xFF) | (unsigned int)head[13]*256);
89 info.height = (int)(((unsigned int)head[14]&0xFF) | (unsigned int)head[15]*256);
90 int imgSize = info.height * info.width;
91
92 unsigned char* data;
93 if((head[2] == 2) && (head[16] == 24))
94 {
95 info.type = RGBA32_RASTER;
96 info.bytesPerPixel = 4;
97 info.bytesPerRow = 4*info.width;
98 data = (unsigned char*) malloc(imgSize*4);
99
100 }
101 else if((head[2] == 3) && (head[16] == 8))
102 {
103 info.type = GRAY_RASTER;
104 info.bytesPerPixel = 1;
105 info.bytesPerRow = info.width;
106 data = (unsigned char*) malloc(imgSize);
107 }
108 else
109 {
110 printf("Unsuported TGA image: ");
111 printf("Its type is %d and it has %d bits per pixel.\n", head[2], head[16]);
112 printf("The image must be uncompressed while having 8 or 24 bits per pixel.\n");
113 fclose(tga_in);
114 exit(12);
115 }
116
117 if(data == NULL)
118 {
119 printf("Unable to allocate memory\n");
120 fclose(tga_in);
121 exit(13);
122 }
123 info.raster = (void*) data;
124
125 //--------------------------------
126 // Read the pixel data
127 //--------------------------------
128
129 // Case of a color image
130 //------------------------
131 //if (info.type == kTGA_COLOR) *****************************
132 if(info.type == RGBA32_RASTER)
133 {
134 // First check if the image is mirrored vertically (a bit setting in the header)
135 if(!(head[17]&0x20))
136 {
137 for(int i = 0; i < info.height; i++)
138 {
139 unsigned char* rowPtr = data + (info.height-i-1)*info.bytesPerRow;
140 for (int j=0; j<info.width; j++)
141 {
142 fread(rowPtr, 3*sizeof(char), 1, tga_in);
143 rowPtr[3] = 0xFF;
144 rowPtr += 4;
145 }
146 }
147 }
148 else
149 {
150 unsigned char* dest = data;
151 for (int i=0; i<info.height; i++)
152 {
153 for (int j=0; j<info.width; j++)
154 {
155 fread(dest, 3*sizeof(char), 1, tga_in);
156 dest+=4;
157 }
158 }
159
160 }
161
162 // tga files store color information in the order B-G-R
163 // we need to swap the Red and Blue components
164 swapRGBA(data, info.height, info.width);
165 }
166
167 // Case of a gray-level image
168 //----------------------------
169 else
170 {
171 // First check if the image is mirrored vertically (a bit setting in the header)
172 if(head[17]&0x20)
173 {
174 unsigned char* ptr = data + imgSize - info.width;
175 for(int i = 0; i < info.height; i++)
176 {
177 fread( ptr, sizeof(char), info.width, tga_in ) ;
178 ptr -= info.width;
179 }
180 }
181 else
182 fread(data, sizeof(char), imgSize, tga_in);
183 }
184
185 fclose( tga_in) ;
186 return info;
187}

190//---------------------------------------------------------------------*
191// Function : writeTGA
192// Description :
193//
194// This function write out an image of type TGA (24-bit color)
195//
196// Return value: Error code (0 = no error)
197//----------------------------------------------------------------------*/
198int writeTGA(char* filePath, RasterImage* info)
199{
200 //--------------------------------
201 // open TARGA output file
202 //--------------------------------
203 FILE* tga_out = fopen(filePath, "wb" );
204 if (tga_out == NULL)
205 {
206 printf("Cannot create image file %s \n", filePath);
207 return 21;
208 }
209
210 // Yes, I know that I tell you over and over that cascading if-else tests
211 // are bad style when testing an integral value, but here only two values
212 // are supported. If I ever add one more I'll use a switch, I promise.
213
214 //------------------------------
215 // Case of a color image
216 //------------------------------
217 if (info->type == RGBA32_RASTER)
218 {
219 //--------------------------------
220 // create the header (TARGA file)
221 //--------------------------------
222 char head[18] ;
223 head[0] = 0 ; // ID field length.
224 head[1] = 0 ; // Color map type.
225 head[2] = 2 ; // Image type: true color, uncompressed.
226 head[3] = head[4] = 0 ; // First color map entry.
227 head[5] = head[6] = 0 ; // Color map lenght.
228 head[7] = 0 ; // Color map entry size.
229 head[8] = head[9] = 0 ; // Image X origin.
230 head[10] = head[11] = 0 ; // Image Y origin.
231 head[13] = (char) (info->width >> 8) ; // Image width.
232 head[12] = (char) (info->width & 0x0FF) ;
233 head[15] = (char) (info->height >> 8) ; // Image height.
234 head[14] = (char) (info->height & 0x0FF) ;
235 head[16] = 24 ; // Bits per pixel.
236// head[17] = 0 ; // Image descriptor bits ;
237 head[17] = 0x20 ; // Image descriptor bits ;
238 fwrite( head, sizeof(char), 18, tga_out );
239
240 unsigned char* data = (unsigned char*) info->raster;
241 for(int i = 0; i < info->height; i++)
242 {
243 unsigned long offset = i*4*info->width;
244 for(int j = 0; j < info->width; j++)
245 {
246 fwrite(&data[offset+2], sizeof(char), 1, tga_out);
247 fwrite(&data[offset+1], sizeof(char), 1, tga_out);
248 fwrite(&data[offset], sizeof(char), 1, tga_out);
249 offset+=4;
250 }
251 }
252
253 fclose( tga_out ) ;
254 }
255 //------------------------------
256 // Case of a gray-level image
257 //------------------------------
258 else if (info->type == GRAY_RASTER)
259 {
260 //--------------------------------
261 // create the header (TARGA file)
262 //--------------------------------
263 char head[18] ;
264 head[0] = 0 ; // ID field length.
265 head[1] = 0 ; // Color map type.
266 head[2] = 3 ; // Image type: gray-level, uncompressed.
267 head[3] = head[4] = 0 ; // First color map entry.
268 head[5] = head[6] = 0 ; // Color map lenght.
269 head[7] = 0 ; // Color map entry size.
270 head[8] = head[9] = 0 ; // Image X origin.
271 head[10] = head[11] = 0 ; // Image Y origin.
272 head[13] = (char) (info->width >> 8) ; // Image width.
273 head[12] = (char) (info->width & 0x0FF) ;
274 head[15] = (char) (info->height >> 8) ; // Image height.
275 head[14] = (char) (info->height & 0x0FF) ;
276 head[16] = 8 ; // Bits per pixel.
277 head[17] = 0 ; // Image descriptor bits ;
278 fwrite( head, sizeof(char), 18, tga_out );
279
280 unsigned char* data = (unsigned char*) info->raster;
281 for(int i = 0; i < info->height; i++)
282 {
283 fwrite(&data[i*info->width], sizeof(char), info->width, tga_out);
284 }
285
286 fclose( tga_out ) ;
287 }
288 else
289 {
290 printf("Image type not supported for output\n");
291 return 22;
292 }
293
294
295
296 return 0;
297}
